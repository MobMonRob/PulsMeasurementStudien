# Dependencies
- python 3.6.7
- opencv 3.4.4
- scipy
- numpy
- matplotlib
- pypylon

# Search for cameras
python video_acquisition.py

# Use video file
python video_acquisition.py [video file]

# Control
- [tab]: switch between cameras
- [p]: start measurement at the roi position
- [space]: start/stop recording
- [arrow_keys]: change position of roi
- [f]: freeze frame
